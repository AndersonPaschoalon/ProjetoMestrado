594  double getRandom(unsigned seed)
595  {
596  MESSER_LOG_INIT(LOG_LEVEL_SF);
597  double retVal = .0;
598 
599  std::ranlux48 generator(seed);
600  if (m_modelName == WEIBULL)
601  {
602  // (alpha, beta) = (shape, scale) = (a, b)
603  std::weibull_distribution<double> distribution(m_param1, m_param2);
604  retVal = distribution(generator);
605  }
606  else if (m_modelName == NORMAL)
607  {
608  // (mu, std) = (mean, standar_deviation)
609  std::normal_distribution<double> distribution(m_param1, m_param2);
610  retVal = distribution(generator);
611  }
612  else if ((m_modelName == EXPONENTIAL)
613  || (m_modelName == EXPONENTIAL_LINEAR_REGRESSION)
614  || (m_modelName == EXPONENTIAL_MEAN))
615  {
616  std::exponential_distribution<double> distribution(m_param1);
617  retVal = distribution(generator);
618  }
619  else if ((m_modelName == PARETO)
620  || (m_modelName == PARETO_LINEAR_REGRESSION)
621  || (m_modelName == PARETO_MAXIMUM_LIKEHOOD))
622  {
623  // No std function offered in C++ standar library, so we use
624  // Exponential to Pareto transformation
625  // https://en.wikipedia.org/wiki/Pareto_distribution#Relation_to_the_exponential_distribution
626  // It says that if X is a Pareto random variable and Y is
627  // a exponential ramdom variable, with rate parameter alpha:
628  // X = x_m * exp(Y)
629  // therefore, after the transformation, we have:
630  // alpha === lambda,
631  std::exponential_distribution<double> distribution(m_param1);
632  retVal = m_param2 * exp(distribution(generator));
633  }
634  else if (m_modelName == CAUCHY)
635  {
636  //http://www.cplusplus.com/reference/random/cauchy_distribution/cauchy_distribution/
637  //https://en.wikipedia.org/wiki/Cauchy_distribution
638  // a === location(x0), b === scale(gamma)
639  std::cauchy_distribution<double> distribution(m_param2, m_param1);
640  retVal = distribution(generator);
641  }
642  else if ((m_modelName == CONSTANT) || (m_modelName == SINGLE_PACKET))
643  {
644  retVal = m_param1;
645  }
646  else if (m_modelName == NO_MODEL)
647  {
648  retVal = 0;
649  }
650  else
651  {
652  fprintf(stderr,
653  "Error: No valid stochastic model (It is `%s)`. Returning default value: `%f`\n",
654  this->strModelName().c_str(), .0);
655  MESSER_CRIT(
656  "<%s> Error: No valid stochastic model (It is `%s)`. Returning default value: `%f`",
657  this->strModelName().c_str(), .0);
658  }
659 
660  MESSER_LOG_END();
661  return (retVal);
662  }
